Copy and paste the code below into assignment 1. Remember, this function needs to be adapted to also accommodate 3 and 4 character hexadecimal color values.

For more information on bitwise operators in Swift, see Swift.org - Advanced Operators (opens new window).

See Also:

  • >> Bitwise Shifting (swift 3) << (opens new window)

For a different approach on converting see Convert between Decimal, Binary and Hexadecimal in Swift (opens new window)

func getRGBAColor(rgba: String) -> (UInt8,UInt8,UInt8,UInt8){
   
    var red: UInt8 = 0
    var green: UInt8 = 0
    var blue: UInt8 = 0
    var alpha: UInt8 = 255
   
    let hexColor = rgba
   
    let scanner = Scanner(string: hexColor)
    var hexNumber: UInt64 = 0
   
    // 8 characters are treated as 8 digit RRGGBBAA values
    if hexColor.count == 8 {
       
        if scanner.scanHexInt64(&hexNumber) {
            red = UInt8((hexNumber & 0xff000000) >> 24)
            green = UInt8((hexNumber & 0x00ff0000) >> 16)
            blue = UInt8((hexNumber & 0x0000ff00) >> 8)
            alpha = UInt8(hexNumber & 0x000000ff)
        }
        // 6 characters are treated as 6 digit RRGGBB values
    } else if hexColor.count == 6 { // assume no alpha value
       
        if scanner.scanHexInt64(&hexNumber) {
            red = UInt8((hexNumber & 0xff0000) >> 16)
            green = UInt8((hexNumber & 0x00ff00) >> 8)
            blue = UInt8((hexNumber & 0x0000ff))
            alpha = 255 // set to max
        }
    }

    return (red,green,blue,alpha) // will return black if failed
}
