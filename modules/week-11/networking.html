# Overview

The NWPathMonitor monitors network connections and their state. To monitor changes:

  1. Create an instance of NWPathMonitor
  2. Define the logic to execute when changes occur in the .UpdatePathHandler method
  3. Create an instance of a dispatch queue to capture updates
  4. Start the monitor and assign the dispatch queue

# Example

import Foundation

// Need the networking package
import Network

class InternetConnection {
    
    // Create an instance of NWPathMonitor
    private let monitor = NWPathMonitor()

    // Our own variables will be updated as changes occur.
    var isConnected = true
    var isExpensive = false
    
    init() {
       
            // this closure gets called whenever the connection status changes
            monitor.pathUpdateHandler = { path in
                
                // .satisfied will be the status if even if there are concurrent
                // concurrent connections with no connectivity
                if path.status == .satisfied{
                    print("Internet Connected")
                    self.isConnected = true // must use self in a closure when referencing a property
                } else {
                    print("Internet Not Connected")
                    self.isConnected = false
                }
                
                // .isExpensive generally indicates a higher cost connection like cellular
                self.isExpensive = path.isExpensive
                print("Is Expensive = \(self.isExpensive)")
            }
            
        // create an instance of a queue for updates to be delivered to    
        let queue = DispatchQueue.global(qos: .background)

        // Start connection monitoring
        monitor.start(queue: queue)
    }
}
