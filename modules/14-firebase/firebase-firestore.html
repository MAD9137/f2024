# ðŸš€ Firestore

Firestore is a flexible, scalable database for mobile, web, and server development. In this lecture, we'll learn how to integrate Firestore into a Swift project, assuming Firebase is already installed, and Authentication is enabled. We'll cover:

  1. Configuring Firestore in your Swift project.
  2. Performing CRUD (Create, Read, Update, Delete) operations.
  3. Querying data.
  4. Handling Firestore listeners for real-time updates.

# 1. Configuring Firestore

# Import Firestore

To start using Firestore, ensure you have the required import statement in your Swift files:

import FirebaseFirestore

# Initialize Firestore

Firestore is initialized by default when Firebase is configured. You can create a shared Firestore instance like this:

# 4. Reading Data

# Fetching All Posts

Use getDocuments to fetch data from a collection:

func fetchPosts() async throws -> [QueryDocumentSnapshot] {
    do {
        let snapshot = try await db.collection("posts")
            .order(by: "timestamp", descending: true)
            .getDocuments()
        
        return snapshot.documents
    } catch {
        print("Error fetching posts: \(error)")
        throw error
    }
}

# Decoding Fetched Data

Hereâ€™s how you can decode the fetched data into a Swift struct:

struct Post: Identifiable, Codable {
    var id: String
    var authorID: String
    var content: String
    var timestamp: Date
    
    init?(from document: QueryDocumentSnapshot) {
        let data = document.data()
        guard 
            let authorID = data["authorID"] as? String,
            let content = data["content"] as? String,
            let timestamp = data["timestamp"] as? Timestamp 
        else { return nil }
        
        self.id = document.documentID
        self.authorID = authorID
        self.content = content
        self.timestamp = timestamp.dateValue()
    }
}

# 7. Real-Time Updates

Firestore supports real-time listeners to keep your app data in sync.

# Listen to Updates on posts

Hereâ€™s how to listen to changes in the posts collection:

func listenToPosts() {
    db.collection("posts")
        .order(by: "timestamp", descending: true)
        .addSnapshotListener { snapshot, error in
            if let error = error {
                print("Error listening for changes: \(error)")
                return
            }
            
            guard let snapshot = snapshot else { return }
            let posts = snapshot.documents.compactMap { Post(from: $0) }
            print("Updated posts: \(posts)")
        }
}

# 8. Querying Data

Firestore supports querying with compound filters.

# Fetch Posts by Author

To fetch posts from a specific author:

func fetchPostsByAuthor(authorID: String) async throws -> [Post] {
    do {
        let snapshot = try await db.collection("posts")
            .whereField("authorID", isEqualTo: authorID)
            .getDocuments()
        
        return snapshot.documents.compactMap { Post(from: $0) }
    } catch {
        print("Error fetching posts by author: \(error)")
        throw error
    }
}

# 9. Subcollections

# Adding Comments to a Post

Hereâ€™s how to add a comment to the comments subcollection of a post:

func addComment(to postID: String, comment: String, authorID: String) async throws {
    let commentData: [String: Any] = [
        "authorID": authorID,
        "comment": comment,
        "timestamp": Timestamp(date: Date())
    ]
    
    do {
        try await db.collection("posts").document(postID)
            .collection("comments").addDocument(data: commentData)
        print("Comment added to post \(postID)")
    } catch {
        print("Error adding comment: \(error)")
        throw error
    }
}

# Fetching Comments for a Post

To fetch all comments for a specific post:

func fetchComments(for postID: String) async throws -> [Comment] {
    do {
        let snapshot = try await db.collection("posts")
            .document(postID)
            .collection("comments")
            .order(by: "timestamp", descending: true)
            .getDocuments()
        
        return snapshot.documents.compactMap { Comment(from: $0) }
    } catch {
        print("Error fetching comments: \(error)")
        throw error
    }
}
